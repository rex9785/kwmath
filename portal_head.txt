<script>
  // ════════ 상태 ════════
  const TOKEN_KEY = 'kwmath_portal_token';
  const PHONE_KEY = 'kwmath_portal_phone';
  let currentToken = '';
  let currentPhone = '';
  let currentStudents = [];
  let selectedStudent = null;

  // ════════ 공통 유틸 ════════
  function $(id) { return document.getElementById(id); }

  function showToast(msg, ms) {
    const t = $('toast');
    t.textContent = msg;
    t.classList.add('show');
    clearTimeout(t._timer);
    t._timer = setTimeout(() => t.classList.remove('show'), ms || 2800);
  }

  function showErr(elId, msg) {
    const el = $(elId);
    el.textContent = msg;
    el.style.display = msg ? 'block' : 'none';
  }

  function formatPhone(el) {
    let v = (el.value || '').replace(/[^0-9]/g, '').slice(0, 11);
    // 8자리 이상일 때만 두 번째 하이픈 추가 (7자리에서 백스페이스로 지울 수 있게)
    if (v.length > 7)       v = v.slice(0,3) + '-' + v.slice(3,7) + '-' + v.slice(7);
    else if (v.length > 3)  v = v.slice(0,3) + '-' + v.slice(3);
    el.value = v;
  }

  function normalizePhone(raw) {
    const d = (raw || '').replace(/[^0-9]/g, '');
    if (d.length === 10) return d.slice(0,3) + '-' + d.slice(3,6) + '-' + d.slice(6);
    if (d.length === 11) return d.slice(0,3) + '-' + d.slice(3,7) + '-' + d.slice(7);
    return null;
  }

  // ════════ 비밀번호 보기 토글 ════════
  function togglePwVisibility(inputId, btn) {
    const inp = $(inputId);
    if (!inp) return;
    if (inp.type === 'password') {
      inp.type = 'text';
      if (btn) { btn.textContent = '🙈'; btn.title = '비밀번호 숨기기'; }
    } else {
      inp.type = 'password';
      if (btn) { btn.textContent = '👁'; btn.title = '비밀번호 보기'; }
    }
  }

  // ════════ 화면 전환 ════════
  function showLoginScreen() {
    $('loginScreen').style.display = 'flex';
    $('mainScreen').style.display = 'none';
    // 로그인 화면 진입 시 입력값 명시적으로 비우기 (자동완성/이전 입력 잔존 방지)
    const ph = $('loginPhone'); if (ph) ph.value = '';
    const pw = $('loginPw');    if (pw) { pw.value = ''; pw.type = 'password'; }
    const er = $('loginErr');   if (er) er.style.display = 'none';
  }
  function showMainScreen() {
    $('loginScreen').style.display = 'none';
    $('mainScreen').style.display = 'block';
  }

  // ════════ 로그인 ════════
  async function doLogin() {
    const phone = normalizePhone($('loginPhone').value);
    const password = $('loginPw').value;
    showErr('loginErr', '');
    if (!phone)    { showErr('loginErr', '휴대폰 번호를 정확히 입력해주세요.'); return; }
    if (!password) { showErr('loginErr', '비밀번호를 입력해주세요.'); return; }

    const btn = $('loginBtn');
    btn.disabled = true; btn.textContent = '로그인 중...';
    try {
      const res = await fetch('/api/auth/login', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ phone, password })
      });
      const data = await res.json();
      if (!res.ok || !data.ok) {
        showErr('loginErr', data.error || '로그인 실패. 다시 시도해주세요.');
        return;
      }
      // 토큰 저장
      currentToken = data.token;
      currentPhone = data.phone;
      currentStudents = Array.isArray(data.students) ? data.students : [];
      localStorage.setItem(TOKEN_KEY, currentToken);
      localStorage.setItem(PHONE_KEY, currentPhone);

      enterMain(data.mustChangePassword);
    } catch (e) {
      showErr('loginErr', '서버 통신 오류: ' + (e.message || '알 수 없는 오류'));
    } finally {
      btn.disabled = false; btn.textContent = '로그인 →';
    }
  }

  // 로그인 성공 후 메인 진입
  function enterMain(mustChangePassword) {
    showMainScreen();
    renderWelcome();
    renderChildSelect();
    updatePushButton();
    showInstallHintIfNeeded();
    if (mustChangePassword) {
      openPasswordModal(true);
    }
  }

  // ════════ 로그아웃 ════════
  function doLogout() {
    if (!confirm('로그아웃하시겠습니까?')) return;
    localStorage.removeItem(TOKEN_KEY);
    localStorage.removeItem(PHONE_KEY);
    currentToken = ''; currentPhone = ''; currentStudents = []; selectedStudent = null;
    $('loginPw').value = '';
    showLoginScreen();
    showToast('로그아웃 되었습니다.');
  }

  // ════════ 환영 메시지 + 자녀 선택 ════════
  function renderWelcome() {
    const first = currentStudents[0];
    const name = first ? first.name : '';
    const role = first ? first.role : '';
    if (name) {
      if (role === 'student') {
        $('welcomeMsg').textContent = name + ' 학생, 환영합니다 👋';
      } else {
        $('welcomeMsg').textContent = name + ' 학생 학부모님 환영합니다 👋';
      }
    } else {
      $('welcomeMsg').textContent = '환영합니다 👋';
    }
    $('welcomePhone').textContent = '📱 ' + currentPhone;
  }

  function renderChildSelect() {
    const wrap = $('childSelectWrap');
    const sel = $('childSelect');
    if (currentStudents.length <= 1) {
      wrap.style.display = 'none';
      selectedStudent = currentStudents[0] || null;
      return;
    }
    wrap.style.display = 'block';
    sel.innerHTML = currentStudents.map((s, i) =>
      `<option value="${i}">${s.name || '—'} (${s.school || ''}${s.grade ? ' · ' + s.grade : ''}${s.className ? ' / ' + s.className : ''})</option>`
    ).join('');
    sel.value = '0';
    selectedStudent = currentStudents[0];
  }

  function onChildChange() {
    const idx = parseInt($('childSelect').value, 10) || 0;
    selectedStudent = currentStudents[idx];
    // 메인 페이지에서 기능 버튼 누를 때 학생 이름이 자동으로 적용되게 할 예정 (Phase 3에서 query string 또는 sessionStorage로 넘김)
    sessionStorage.setItem('kwmath_selected_student', JSON.stringify(selectedStudent));
  }

  let pwModalForced = false; // 강제 모드 (mustChangePassword)일 때 취소 불가

  function openPasswordModal(forced) {
    pwModalForced = !!forced;
    $('pwOld').value = '';
    $('pwNew').value = '';
    $('pwNew2').value = '';
    showErr('pwErr', '');
    if (forced) {

      $('pwModalSub').textContent = '안전한 사용을 위해 비밀번호를 변경해주세요. (4자 이상, 처음엔 0000)';

    } else {
      $('pwModalTitle').textContent = '🔑 비밀번호 변경';
      $('pwModalSub').textContent = '새 비밀번호를 설정하세요. (4자 이상)';

    }
    $('pwModal').classList.add('show');

  }

  function closePasswordModal() {
    if (pwModalForced) return; // 강제일 땐 못 닫음
    $('pwModal').classList.remove('show');
  }

  async function doChangePassword() {
    const oldPw  = $('pwOld').value;
    const newPw  = $('pwNew').value;
    const newPw2 = $('pwNew2').value;
    showErr('pwErr', '');
    if (!oldPw) { showErr('pwErr', '현재 비밀번호를 입력해주세요.'); return; }
    if (!newPw || newPw.length < 4) { showErr('pwErr', '새 비밀번호는 4자 이상이어야 합니다.'); return; }
    if (newPw !== newPw2) { showErr('pwErr', '새 비밀번호가 일치하지 않습니다.'); return; }
    if (oldPw === newPw) { showErr('pwErr', '새 비밀번호가 기존과 동일합니다.'); return; }

    try {
      const res = await fetch('/api/auth/change-password', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json', 'Authorization': 'Bearer ' + currentToken },
        body: JSON.stringify({ oldPassword: oldPw, newPassword: newPw })
      });
      const data = await res.json();
      if (!res.ok || !data.ok) {
        showErr('pwErr', data.error || '변경 실패');
        return;
      }
      pwModalForced = false;
      $('pwModal').classList.remove('show');
      showToast('✅ 비밀번호가 변경되었습니다.');
    } catch (e) {
      showErr('pwErr', '오류: ' + (e.message || '알 수 없는 오류'));
    }
  }

  // ════════ 푸쉬 알림 (PWA) ════════
  function urlBase64ToUint8Array(b64) {
    const padding = '='.repeat((4 - b64.length % 4) % 4);
    const base64 = (b64 + padding).replace(/-/g, '+').replace(/_/g, '/');
    const raw = atob(base64);
    const out = new Uint8Array(raw.length);
    for (let i = 0; i < raw.length; i++) out[i] = raw.charCodeAt(i);
    return out;
  }

  async function isPushSubscribed() {
    if (!('serviceWorker' in navigator) || !('PushManager' in window)) return false;
    try {
      const reg = await navigator.serviceWorker.ready;
      const sub = await reg.pushManager.getSubscription();
      return !!sub;
    } catch (_) { return false; }
  }

  async function toggleEnablePush() {
    if (!('serviceWorker' in navigator) || !('PushManager' in window)) {
      showToast('이 브라우저는 푸쉬 알림을 지원하지 않습니다.');
      return;
    }
    if (Notification.permission === 'denied') {
      showToast('알림 권한이 거부됐어요. 브라우저 설정에서 허용해주세요.', 4000);
      return;
    }
    try {
      // 이미 구독 중이면 해제
      const reg = await navigator.serviceWorker.ready;
      const existing = await reg.pushManager.getSubscription();
      if (existing) {
        await existing.unsubscribe();
        // 서버에서도 해제 (구현돼있다면)
        try {
          await fetch('/api/push-subscribe', {
            method: 'DELETE',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ userId: currentPhone, endpoint: existing.endpoint })
          });
        } catch (_) {}
        showToast('알림이 꺼졌습니다.');
        updatePushButton();
        return;
      }

      // 권한 요청
      const perm = await Notification.requestPermission();
      if (perm !== 'granted') {
        showToast('알림 권한이 필요합니다.');
        return;
      }
      // VAPID 공개키
      const vRes = await fetch('/api/push-vapid-public');
      const vData = await vRes.json();
      if (!vData.publicKey) {
        showToast('푸쉬 서버 설정이 안 되어있어요. 관우T께 문의해주세요.', 4000);
        return;
      }
      // 구독
      const sub = await reg.pushManager.subscribe({
        userVisibleOnly: true,
        applicationServerKey: urlBase64ToUint8Array(vData.publicKey),
      });
      // 서버 저장
      await fetch('/api/push-subscribe', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ userId: currentPhone, subscription: sub.toJSON() }),
      });
      showToast('🔔 알림이 활성화됐어요!');
      updatePushButton();
    } catch (e) {
      showToast('알림 설정 오류: ' + (e.message || '알 수 없는 오류'), 4000);
    }
  }
